	slot = 2
	generic = no
	ai = yes
	potential_on_load = {
		has_dlc = "King of Kings"
	}
	potential = {
		NOT = {
			map_setup = map_setup_random
		}
		has_dlc = "King of Kings"
		tag = BAL
	}
	has_country_shield = yes
	bal_conquer_sindh = {
		icon = mission_conquer_sindh_dlh
		position = 1
		required_missions = {
		}
		provinces_to_highlight = {
			OR = {
				area = sindh_area
				area = northern_sindh_area
			}
			NOT = {
				country_or_non_sovereign_subject_holds = root
			}
		}
		trigger = {
			owns_all_provinces = {
				area = sindh_area
			}
			owns_all_provinces = {
				area = northern_sindh_area
			}
		}
		effect = {
			add_permanent_claim_multi_provinces = {
				target = west_india_region
			}
		}
	}
}
